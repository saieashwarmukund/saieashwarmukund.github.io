import plotly.express as px
import pandas as pd
import datetime
import js

# Set the current date in the format you want
present_dt = datetime.datetime.now()
present = str(present_dt.strftime("%Y-%m"))

# Data
data = [['Education', '2016-09', '2020-12', "B.S. in Data Science", "University of California San Diego"],
        ['Experience', '2018-10', '2019-06', "Research Assistant/Intern", "Computational Neural Data and Dynamics Lab"],
        ['Experience', '2019-06', '2019-08', "Business Intelligence Internship", "Alliant Insurance"],
        ['Experience', '2021-04', '2022-09', "Business Intelligence Analyst", "Liftoff"],
        ['Experience', '2023-03', present, "Technical Analyst", "Coforge"],
        ['Project', '2018-06', '2018-09', "BhajanBook Web Application", "Sai Center of Tustin"],
        ['Project', '2020-04', '2020-06', "Red Means Go", "University of California San Diego"]]

df = pd.DataFrame(data, columns=['Type', 'Start', 'End', 'Title', 'Location'])


# Create the timeline plot
fig = px.timeline(
    df,
    x_start="Start",
    x_end="End",
    y="Type",
    color="Title",
    custom_data=['Title', 'Location', 'Start', 'End']
)

# Reverse the y-axis so tasks are listed from top to bottom
fig.update_yaxes(autorange="reversed")

# Update the hover template to display in the desired format
fig.update_traces(
    hovertemplate="%{customdata[0]} | " +
                    "%{customdata[1]} | " +
                    "%{base|%b %Y} - %{x|%b %Y}"
)


fig.update_layout(
    autosize=True,
    minreducedwidth=240,
    minreducedheight=240,
    height=360,
    hoverlabel=dict(
        bgcolor="white",
        font_size=16,
        font_family="Arial"
    ),
    title=None,
    showlegend=False,
    xaxis_title=None,
    yaxis_title=None
)

#fig.show()
js.plot(fig.to_json(), "timeline-output")
